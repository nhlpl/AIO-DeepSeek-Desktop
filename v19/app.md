## 1. Monads – `core/utils/monad.mbt`

```moonbit
// Monad trait and bind operator for Result and Async
trait Monad[M[_]] {
  fn return_(a: A) -> M[A]
  fn bind(m: M[A], f: (A) -> M[B]) -> M[B]
}

fn >>=[M[_], A, B](m: M[A], f: (A) -> M[B]) -> M[B] with Monad[M] {
  Monad::bind(m, f)
}

// Result monad
impl Monad[Result] for Result {
  fn return_(a: A) -> Result[A] { Ok(a) }
  fn bind(m: Result[A], f: (A) -> Result[B]) -> Result[B] {
    match m { Ok(a) => f(a), Err(e) => Err(e) }
  }
}

// Async monad (simplified – assume pure returns)
impl Monad[Async] for Async {
  fn return_(a: A) -> Async[A] { Async::pure(a) }
  fn bind(m: Async[A], f: (A) -> Async[B]) -> Async[B] { m.and_then(f) }
}
```

Now error handling becomes linear:
```moonbit
api_call() >>= process >>= save
```

## 2. Lenses – `core/utils/lens.mbt`

```moonbit
struct Lens[S, A] {
  get: (S) -> A
  set: (S, A) -> S
}

fn Lens::compose[S, A, B](self: Lens[S, A], other: Lens[A, B]) -> Lens[S, B] {
  Lens {
    get: fn(s) { other.get(self.get(s)) },
    set: fn(s, b) { self.set(s, other.set(self.get(s), b)) }
  }
}

fn Lens::over[S, A](self: Lens[S, A], s: S, f: (A) -> A) -> S {
  self.set(s, f(self.get(s)))
}

// Example: lens for a field of a record (manually defined; could be generated by macro)
fn satisfaction_lens() -> Lens[Observer, Float64] {
  Lens{
    get: fn(o) { o.satisfaction },
    set: fn(o, v) { Observer{ ..o, satisfaction: v } }
  }
}
```

Using a lens:
```moonbit
let new_obs = satisfaction_lens().set(old_obs, 0.9)
```

## 3. Recursion Schemes – `core/utils/recursion.mbt`

```moonbit
// A simple binary tree (for inverted index)
enum Tree[A] { Leaf(A), Node(Array[Tree[A]]) }

// Catamorphism (fold)
fn Tree::cata[A, B](self: Tree[A], alg: (A, Array[B]) -> B) -> B {
  match self {
    Leaf(a) => alg(a, [])
    Node(children) => {
      let child_vals = children.map(fn(t) { t.cata(alg) })
      // Note: the algebra also receives the node's own A value, but here we have no node value.
      // For a leaf, we have A; for internal nodes, we may have no value. Adjust type accordingly.
      // We'll use a variant: Leaf(A) and Node(children) – internal node has no own value.
      // So algebra for internal node only receives children results.
      alg(??, child_vals) // Dummy placeholder – refactor for your tree structure.
    }
  }
}

// Better: separate algebra for leaf and node
type TreeAlgebra[A, B] = (leaf: (A) -> B, node: (Array[B]) -> B)

fn Tree::fold[A, B](self: Tree[A], alg: TreeAlgebra[A, B]) -> B {
  match self {
    Leaf(a) => alg.leaf(a)
    Node(children) => alg.node(children.map(fn(t) { t.fold(alg) }))
  }
}
```

Now summing importance becomes:
```moonbit
let total = tree.fold(leaf: fn(m) { m.importance }, node: fn(ch) { ch.sum() })
```

## 5. Algebraic Effects Simulation – `core/plugins/plugin_host.mbt`

We simulate effects using a free monad over a coproduct of effect functors.

```moonbit
enum HostEffect {
  PlaySound(String)
  TriggerHaptic(String)
}

type Free[F, A] = Pure(A) | Impure(F, (Any) -> Free[F, A])

fn pure[F, A](a: A) -> Free[F, A] { Pure(a) }
fn impure[F, A](e: F, k: (Any) -> Free[F, A]) -> Free[F, A] { Impure(e, k) }

// Specific operations
fn play_sound(path: String) -> Free[HostEffect, Unit] {
  impure(PlaySound(path), fn(_) { pure(()) })
}
fn trigger_haptic(pattern: String) -> Free[HostEffect, Unit] {
  impure(TriggerHaptic(pattern), fn(_) { pure(()) })
}

// Interpreter
fn run_host(prog: Free[HostEffect, Unit], permissions: Set[String]) -> Unit {
  match prog {
    Pure(_) => ()
    Impure(PlaySound(path), k) =>
      if permissions.contains("play_sound") {
        host_play_sound(path)
        run_host(k(()), permissions)
      } else {
        log_warning("Plugin not allowed to play sound")
      }
    Impure(TriggerHaptic(pattern), k) =>
      if permissions.contains("haptic") {
        host_trigger_haptic(pattern)
        run_host(k(()), permissions)
      } else {
        log_warning("Plugin not allowed to trigger haptic")
      }
  }
}
```

Plugin code is pure and declarative:
```moonbit
let prog = play_sound("click.wav") >>= fn(_) { trigger_haptic("bump") }
run_host(prog, permissions)
```
